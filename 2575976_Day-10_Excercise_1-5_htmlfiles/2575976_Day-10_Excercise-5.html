christmas_movies_usa = release_dates_df[(release_dates_df['title'].str.contains("Christmas", case=False, na=False)) &

christmas_movies_usa['month'] = pd.to_datetime(christmas_movies_usa['date']).dt.month

plt.title('Monthly Distribution of Movies with "Christmas" in Title (USA)')
plt.xlabel('Month')
plt.ylabel('Number of Movies')
plt.xticks(range(12), month_names, rotation=45)
plt.show()

In which months do films with Judi Dench tend to be released in the USA?¶

In [12]:
judi_dench_movies = cast_df[cast_df['name'] == 'Judi Dench']

judi_dench_movies = pd.merge(judi_dench_movies, release_dates_df, on=['title', 'year'])

judi_dench_movies['date'] = pd.to_datetime(judi_dench_movies['date'], errors='coerce')
judi_dench_movies = judi_dench_movies.dropna(subset=['date'])

judi_dench_movies_usa = judi_dench_movies[judi_dench_movies['country'] == 'USA']

judi_dench_movies_usa = judi_dench_movies_usa.copy()
judi_dench_movies_usa['release_month'] = judi_dench_movies_usa['date'].dt.month

release_month_counts = judi_dench_movies_usa['release_month'].value_counts().sort_index()

plt.figure(figsize=(10, 6))
release_month_counts.plot(kind='bar', color='skyblue')
plt.title("Judi Dench Movies Released in the USA by Month")
plt.xlabel("Month")
plt.ylabel("Number of Movies")
plt.xticks(rotation=45)
plt.show()
